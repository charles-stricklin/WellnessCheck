import Foundation
import CoreLocation
import CoreMotion

class VitalSignsTracker: NSObject, CLLocationManagerDelegate {
	static let shared = VitalSignsTracker()

	private let pedometer = CMPedometer()
	private let locationManager = CLLocationManager()

	private var lastKnownSteps: Int = 0
	private var lastKnownLocation: CLLocation?

	private let inactivityThreshold: TimeInterval = 6 * 3600 // 6 hours

	private override init() {
		super.init()
		startTracking()
	}

	private func startTracking() {
		pedometer.startUpdates(from: Date()) { [weak self] data, _ in
			if let steps = data?.numberOfSteps.intValue {
				self?.lastKnownSteps = steps
			}
		}

		locationManager.delegate = self
		locationManager.requestAlwaysAuthorization()
		locationManager.allowsBackgroundLocationUpdates = true
		locationManager.startMonitoringSignificantLocationChanges()
	}

	func evaluateProofOfLife() {
		let now = Date()
		let lastActive = WellnessCheckViewModel.shared?.lastActive ?? now
		let inactiveFor = now.timeIntervalSince(lastActive)

		let moved = hasMovedSignificantly()
		let stepped = lastKnownSteps > 0

		if inactiveFor > inactivityThreshold && !moved && !stepped {
			let msg = "No proof of life: \(Int(inactiveFor / 60)) min, 0 steps, no location change"
			AlertLogger.shared.logEvent(msg, location: lastKnownLocation?.coordinate, source: "VitalSigns")
		}
	}

	private func hasMovedSignificantly() -> Bool {
		guard let location = locationManager.location else { return false }
		if let last = lastKnownLocation {
			let distance = location.distance(from: last)
			return distance > 10 // meters
		} else {
			lastKnownLocation = location
			return false
		}
	}

	func locationManager(_ manager: CLLocationManager, didUpdateLocations locations: [CLLocation]) {
		if let latest = locations.last {
			lastKnownLocation = latest
		}
	}
}

import Foundation
import SwiftUI
import Combine

final class IdleManager: ObservableObject {
	static let shared = IdleManager()

	@Published var isIdle: Bool = false

	private var idleTimer: Timer?
	private let idleDuration: TimeInterval = BuildConfig.idleTimeout

	private init() {
		startIdleTimer()
		setupNotifications()
	}

	func resetIdleTimer() {
		idleTimer?.invalidate()
		isIdle = false
		startIdleTimer()
	}

	private func startIdleTimer() {
		idleTimer = Timer.scheduledTimer(withTimeInterval: idleDuration, repeats: false) { [weak self] _ in
			self?.isIdle = true
		}
	}

	private func setupNotifications() {
		NotificationCenter.default.addObserver(forName: UIApplication.willResignActiveNotification, object: nil, queue: .main) { _ in
			self.idleTimer?.invalidate()
		}

		NotificationCenter.default.addObserver(forName: UIApplication.didBecomeActiveNotification, object: nil, queue: .main) { _ in
			self.isIdle = false
			self.resetIdleTimer()
		}
	}
}

import SwiftUI

extension Color {
	/// Initializes a `Color` from a hex string like "#0e3355" or "#d8edf7ff"
	init(hex: String) {
		var hexSanitized = hex.trimmingCharacters(in: .whitespacesAndNewlines).uppercased()
		hexSanitized = hexSanitized.replacingOccurrences(of: "#", with: "")

		var rgb: UInt64 = 0
		Scanner(string: hexSanitized).scanHexInt64(&rgb)

		let r, g, b, a: Double
		switch hexSanitized.count {
		case 6:
			r = Double((rgb & 0xFF0000) >> 16) / 255.0
			g = Double((rgb & 0x00FF00) >> 8) / 255.0
			b = Double(rgb & 0x0000FF) / 255.0
			a = 1.0
		case 8:
			r = Double((rgb & 0xFF000000) >> 24) / 255.0
			g = Double((rgb & 0x00FF0000) >> 16) / 255.0
			b = Double((rgb & 0x0000FF00) >> 8) / 255.0
			a = Double(rgb & 0x000000FF) / 255.0
		default:
			r = 0; g = 0; b = 0; a = 1
		}

		self.init(red: r, green: g, blue: b, opacity: a)
	}
}

import Foundation
import SwiftKeychainWrapper

final class PINManager {
	
	static let shared = PINManager()
	
	private let pinKey = "caregiverPIN"
	
	private init() {}

	/// Checks if a PIN is already set
	var isPINSet: Bool {
		return KeychainWrapper.standard.hasValue(forKey: pinKey)
	}
	
	/// Sets the caregiver PIN securely
	/// - Parameter pin: A 6-digit string
	/// - Returns: Bool indicating success
	@discardableResult
	func setPIN(_ pin: String) -> Bool {
		guard pin.count == 6, pin.allSatisfy(\.isNumber) else {
			return false
		}
		return KeychainWrapper.standard.set(pin, forKey: pinKey)
	}

	/// Verifies entered PIN against stored value
	/// - Parameter input: Entered 6-digit code
	/// - Returns: Bool indicating match
	func verifyPIN(_ input: String) -> Bool {
		return KeychainWrapper.standard.string(forKey: pinKey) == input
	}
	
	/// Clears the saved PIN
	func clearPIN() {
		KeychainWrapper.standard.removeObject(forKey: pinKey)
	}
	
	/// Retrieves stored PIN (only for testing, DO NOT expose in prod)
	func debugStoredPIN() -> String? {
		#if DEBUG
		return KeychainWrapper.standard.string(forKey: pinKey)
		#else
		return nil
		#endif
	}
}

//
//  MotionMonitor.swift
//  WellnessCheck
//
//  Created by Charles Stricklin on 6/26/2025
//  Version 0.002a Jul 04, 2025 at 09:50 PM
//  Added:
//  - DeviceMotion + Pedometer integration
//  - Fall detection using acceleration magnitude
//  - Published step count and fall status
//  Updated:
//  - Added debug logs and availability checks
//

import Foundation
import CoreMotion
import Combine

class MotionMonitor: ObservableObject {
	@Published var fallDetected: Bool = false
	@Published var gaitWarning: Bool = false
	@Published var stepCount: Int = 0

	private let motionManager = CMMotionManager()
	private let pedometer = CMPedometer()
	private let motionQueue = OperationQueue()

	init() {
		setupMotion()
		setupPedometer()
	}

	private func setupMotion() {
		guard motionManager.isDeviceMotionAvailable else {
			print("⚠️ Device motion not available")
			return
		}

		motionManager.deviceMotionUpdateInterval = 0.1
		motionManager.startDeviceMotionUpdates(to: motionQueue) { [weak self] motion, _ in
			guard let self = self, let data = motion else { return }
			self.detectFall(with: data)
		}

		print("🧠 Device motion updates started")
	}

	private func setupPedometer() {
		guard CMPedometer.isStepCountingAvailable() else {
			print("⚠️ Step counting not available")
			return
		}

		pedometer.startUpdates(from: Date()) { [weak self] data, error in
			DispatchQueue.main.async {
				if let steps = data?.numberOfSteps.intValue {
					self?.stepCount = steps
					print("👟 Step count updated: \(steps)")
				} else if let err = error {
					print("❗ Step tracking error: \(err.localizedDescription)")
				}
			}
		}

		print("👣 Step tracking started")
	}

	private func detectFall(with data: CMDeviceMotion) {
		let accel = sqrt(
			data.userAcceleration.x * data.userAcceleration.x +
			data.userAcceleration.y * data.userAcceleration.y +
			data.userAcceleration.z * data.userAcceleration.z
		)

		print("📉 Acceleration: \(accel)")

		if accel < 0.5 {
			DispatchQueue.main.async {
				if !self.fallDetected {
					self.fallDetected = true
					print("🛑 Fall detected!")
				}
			}
		}
	}
}

import Foundation
import FirebaseFirestore
import CoreLocation

class AlertLogger {
	static let shared = AlertLogger()
	private let fileManager = FileManager.default
	private let logsDirectory: URL
	private let firestore = Firestore.firestore()

	private init() {
		logsDirectory = fileManager.urls(for: .documentDirectory, in: .userDomainMask)[0]
		purgeOldLogs(maxAgeDays: 3)
	}

	func logEvent(_ message: String, location: CLLocationCoordinate2D?, source: String) {
		let timestamp = ISO8601DateFormatter().string(from: Date())
		var logEntry = "[\(timestamp)] [\(source)] \(message)"
		if let loc = location {
			logEntry += " @ (Lat: \(loc.latitude), Lon: \(loc.longitude))"
		}

		writeToFile(logEntry)
		writeToFirebase(message, location: location, source: source)
	}

	private func writeToFile(_ logEntry: String) {
		let fileName = "log-\(DateFormatter.yyyyMMdd.string(from: Date())).txt"
		let fileURL = logsDirectory.appendingPathComponent(fileName)

		if !fileManager.fileExists(atPath: fileURL.path) {
			fileManager.createFile(atPath: fileURL.path, contents: nil)
		}

		if let handle = try? FileHandle(forWritingTo: fileURL) {
			handle.seekToEndOfFile()
			if let data = (logEntry + "\n").data(using: .utf8) {
				handle.write(data)
			}
			handle.closeFile()
		}
	}

	private func writeToFirebase(_ message: String, location: CLLocationCoordinate2D?, source: String) {
		var data: [String: Any] = [
			"message": message,
			"timestamp": Timestamp(date: Date()),
			"source": source
		]
		if let loc = location {
			data["latitude"] = loc.latitude
			data["longitude"] = loc.longitude
		}

		firestore.collection("logs").addDocument(data: data)
	}

	private func purgeOldLogs(maxAgeDays: Int) {
		guard let files = try? fileManager.contentsOfDirectory(at: logsDirectory, includingPropertiesForKeys: nil) else { return }
		let cutoffDate = Calendar.current.date(byAdding: .day, value: -maxAgeDays, to: Date())!

		for file in files where file.lastPathComponent.hasPrefix("log-") {
			let dateStr = file.lastPathComponent.replacingOccurrences(of: "log-", with: "").replacingOccurrences(of: ".txt", with: "")
			if let fileDate = DateFormatter.yyyyMMdd.date(from: dateStr), fileDate < cutoffDate {
				try? fileManager.removeItem(at: file)
			}
		}
	}
}

extension DateFormatter {
	static let yyyyMMdd: DateFormatter = {
		let df = DateFormatter()
		df.dateFormat = "yyyy-MM-dd"
		return df
	}()
}

import Foundation
import CoreMotion
import Combine
import CoreLocation

class WellnessCheckViewModel: NSObject, ObservableObject, CLLocationManagerDelegate {
	
	static var shared: WellnessCheckViewModel?

	@Published var stepCount: Int = 0
	@Published var inactivityTimeout: TimeInterval
	@Published var isInactive: Bool = false
	@Published var currentLocation: CLLocationCoordinate2D?
	@Published var fallDetected: Bool = false
	@Published var gaitWarning: Bool = false
	@Published var lastActive: Date = Date()
	
	private let pedometer = CMPedometer()
	private var inactivityTimer: Timer?
	private let inactivityKey = "inactivityTimeout"
	private let defaultTimeout: TimeInterval = 60
	private let locationManager = CLLocationManager()
	private let motionMonitor = MotionMonitor()
	private var cancellables = Set<AnyCancellable>()
	
	override init() {
		let savedTimeout = UserDefaults.standard.double(forKey: inactivityKey)
		inactivityTimeout = savedTimeout > 0 ? savedTimeout : defaultTimeout
		super.init()
		WellnessCheckViewModel.shared = self
		startStepTracking()
		setupLocationTracking()
		resetInactivityTimer()
		bindMotionMonitor()
		observeUserInteraction()
		observeLifecycle()
	}

	private func observeUserInteraction() {
		NotificationCenter.default.publisher(for: .userDidInteract)
			.receive(on: RunLoop.main)
			.sink { [weak self] _ in
				self?.userDidInteract()
			}
			.store(in: &cancellables)
	}

	private func observeLifecycle() {
		AppLifecycleMonitor.shared.$lastBecameActive
			.compactMap { $0 }
			.sink { [weak self] _ in
				self?.userDidInteract()
			}
			.store(in: &cancellables)
	}
	
	private func bindMotionMonitor() {
		motionMonitor.$fallDetected
			.receive(on: RunLoop.main)
			.sink { [weak self] detected in
				guard let self = self, detected else { return }
				AlertLogger.shared.logEvent("Fall detected", location: self.currentLocation, source: "CareRecipient")
				self.fallDetected = true
			}
			.store(in: &cancellables)

		motionMonitor.$gaitWarning
			.receive(on: RunLoop.main)
			.sink { [weak self] warning in
				guard let self = self, warning else { return }
				AlertLogger.shared.logEvent("Gait warning: possible walking detected", location: self.currentLocation, source: "CareRecipient")
				self.gaitWarning = true
			}
			.store(in: &cancellables)
	}
	
	func startStepTracking() {
		guard CMPedometer.isStepCountingAvailable() else {
			print("🚫 Step counting not available.")
			return
		}
		
		pedometer.startUpdates(from: Date()) { [weak self] data, _ in
			DispatchQueue.main.async {
				self?.stepCount = data?.numberOfSteps.intValue ?? 0
			}
		}
	}
	
	func resetInactivityTimer() {
		inactivityTimer?.invalidate()
		isInactive = false
		
		inactivityTimer = Timer.scheduledTimer(withTimeInterval: inactivityTimeout, repeats: false) { [weak self] _ in
			self?.isInactive = true
		}
	}
	
	func userDidInteract() {
		lastActive = Date()
		resetInactivityTimer()
	}
	
	func saveInactivityTimeout(_ duration: TimeInterval) {
		inactivityTimeout = duration
		UserDefaults.standard.set(duration, forKey: inactivityKey)
		resetInactivityTimer()
	}
	
	func setupLocationTracking() {
		locationManager.delegate = self
		locationManager.desiredAccuracy = kCLLocationAccuracyBest
		locationManager.requestWhenInUseAuthorization()
		locationManager.allowsBackgroundLocationUpdates = true
		locationManager.startUpdatingLocation()
	}
	
	func locationManager(_ manager: CLLocationManager, didUpdateLocations locations: [CLLocation]) {
		if let location = locations.last {
			DispatchQueue.main.async {
				self.currentLocation = location.coordinate
			}
		}
	}
}
